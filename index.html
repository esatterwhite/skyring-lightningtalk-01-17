10 Lines of Code ( literally )
        
        const timers = Object.create(null);
        function set(id, amount, fn) {
          if (!timers[id]) {
            timers[id] = setTimeout( fn, amount ).unref();
          }
        }

        function cancel(id) {
          const ref = timers[id];
          return ref && clearTimeout( ref );
        }
